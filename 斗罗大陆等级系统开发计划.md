## 1. 项目概述

### 1.1 项目背景

基于《斗罗大陆》小说的等级体系，为现有的英语单词拼写游戏添加游戏化升级系统，提升用户学习动力。

### 1.2 核心功能

- **等级系统：** 10 个大阶、100 个小级的斗罗大陆等级体系
- **魂力修炼：** 通过单词拼写练习获得魂力值
- **日常任务：** 每日修炼任务，获取额外魂力
- **副本挑战：** 9 个难度递增的副本任务
- **魂骨系统：** 25 种魂骨，提供僵尸对战增益
- **魂骨图鉴：** 收集、鉴定、装配魂骨

### 1.3 现有代码基础

**技术栈：**
- 前端：原生 JavaScript + HTML5 + CSS3
- 数据库：IndexedDB (Dexie.js)
- UI 框架：无（自定义样式）
- 图标：Font Awesome

**已有功能：**
- ✅ 单词练习模式（拼写练习）
- ✅ 句子填空练习模式
- ✅ 僵尸对战游戏机制
- ✅ 积分系统（现有积分可转换为魂力）
- ✅ 本地数据存储（words, books, practiceScores）
- ✅ 玩家等级系统（现有王者荣耀体系，需替换）

**魂力值计算：**
```javascript
function calculateSpiritLevel(totalSpiritPower) {
  let cumulativeRequired = 0;
  
  for (let tier of SPIRIT_LEVEL_SYSTEM.tiers) {
    const tierTotal = tier.levels * tier.spiritPowerPerLevel;
    if (totalSpiritPower < cumulativeRequired + tierTotal) {
      const remainingInTier = totalSpiritPower - cumulativeRequired;
      const levelInTier = Math.floor(remainingInTier / tier.spiritPowerPerLevel) + 1;
      const progressInLevel = remainingInTier % tier.spiritPowerPerLevel;
      
      return {
        tier: tier.name,
        tierId: tier.id,
        level: (tier.id - 1) * 10 + levelInTier,
        levelInTier: levelInTier,
        title: tier.titles[levelInTier - 1],
        progress: progressInLevel / tier.spiritPowerPerLevel * 100,
        nextLevelPower: tier.spiritPowerPerLevel - progressInLevel
      };
    }
    cumulativeRequired += tierTotal;
  }
  
  // 达到满级
  return {
    tier: '封号斗罗',
    tierId: 10,
    level: 100,
    levelInTier: 10,
    title: '神级斗罗',
    progress: 100,
    nextLevelPower: 0
  };
}
```

**副本挑战流程：**
```javascript
async function startDungeonChallenge(dungeonId) {
  const dungeon = DUNGEONS.find(d => d.id === dungeonId);
  const progress = await getDungeonProgress(playerId, dungeonId);
  
  // 检查今日次数
  if (progress.todayCount >= 3 && !progress.isBreakthrough) {
    showError('今日挑战次数已达上限');
    return;
  }
  
  // 进入挑战模式
  state.isDungeonMode = true;
  state.currentDungeon = dungeon;
  
  // 启动练习（根据副本要求）
  if (dungeon.requirements.mode === 'spelling') {
    await startSpellingPractice(dungeon.requirements.wordCount);
  } else {
    await startSentencePractice(dungeon.requirements.wordCount);
  }
}

async function completeDungeon(success, practiceData) {
  const dungeon = state.currentDungeon;
  
  if (success && practiceData.accuracy >= dungeon.requirements.accuracyRequired) {
    // 计算奖励
    const basePoints = practiceData.score;
    const bonusPoints = Math.floor(basePoints * dungeon.pointsMultiplier);
    
    // 发放魂力
    await addSpiritPower(bonusPoints, 'dungeon');
    
    // 魂骨掉落判定
    if (Math.random() < dungeon.soulBoneDropRate) {
      const soulBone = await generateSoulBone();
      await awardSoulBone(soulBone);
      showSoulBoneAward(soulBone);
    }
    
    // 更新副本进度
    await updateDungeonProgress(dungeon.id, true);
    
    // 如果是首次突破
    if (state.isBreakthrough) {
      await completeBreakthrough(dungeon.id);
    }
    
    return { success: true, points: bonusPoints };
  } else {
    // 挑战失败
    await updateDungeonProgress(dungeon.id, false);
    return { success: false };
  }
}
```

function calculatePlayerStats(equippedBones) {

  // 累加魂骨属性
  equippedBones.forEach(bone => {
    if (bone.isEquipped && bone.isIdentified) {
      switch (bone.attributeType) {

          stats.health += bone.attributeValue;
          break;
        case 'defense':
          stats.defense += bone.attributeValue;
          break;
        case 'dodge':
          stats.dodge += bone.attributeValue;
          break;
        case 'knockback':
          stats.knockback += bone.attributeValue;
          break;
        case 'slow':
          // 僵尸减速，单独存储
          stats.zombieSlow = bone.attributeValue;
          break;
      }
    }
  });
  
  return stats;
}
